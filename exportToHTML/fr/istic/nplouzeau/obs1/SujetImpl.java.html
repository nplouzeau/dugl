11   public class SujetImpl<T> implements Sujet<T> { 
12       private List<Observateur<T>> abonnés; 
13       private T valeur;
 
14    
15       public SujetImpl() { 
16           abonnés = new ArrayList<>(); 
17       }
 
18    
19       /** 
20        * Ajoute un observateur o, à condition que o ne soit pas déjà abonné à this 
21        * 
22        * @param o L'observateur à ajouter 
23        * @throws IllegalArgumentException si o est null ou o est déjà abonné à this 
24        */ 
25       @Override 
26       public void abonner(Observateur<T> o) throws IllegalArgumentException { 
27           if (abonnés.contains(o)) { 
28               throw new IllegalArgumentException("o est déjà abonné"); 
29           } 
30           abonnés.add(o); 
31       }
 
32    
33       /** 
34        * Retire un observateur o de la liste des abonnés à this, 
35        * à condition que 
36        * o soit abonné 
37        * 
38        * @param o L'observateur à retirer 
39        * @throws IllegalArgumentException si o est null ou o n'est pas abonné à this 
40        */ 
41       @Override 
42       public void désabonner(Observateur<T> o) throws IllegalArgumentException { 
43           if (!abonnés.contains(o)) { 
44               throw new IllegalArgumentException("o n'est pas abonné"); 
45           } 
46           abonnés.remove(o); 
47       }
 
48    
49       /** 
50        * Lit la valeur du sujet 
51        * 
52        * @return valeur du sujet 
53        */ 
54       @Override 
55       public T getValeur() { 
56           return valeur; 
57       }
 
58    
59       /** 
60        * Indique si un observateur est abonné au sujet this 
61        * 
62        * @param o l'observateur à tester 
63        * @return true ssi o est déjà abonné 
64        * @throws IllegalArgumentException si o est null 
65        */ 
66       @Override 
67       public boolean estAbonné(Observateur<T> o) { 
68           if (o == null) { 
69               throw new IllegalArgumentException("o est null"); 
70           } 
71           return abonnés.contains(o); 
72       }
 
73    
74       /** 
75        * Change la valeur du sujet 
76        * 
77        * @param v la nouvelle valeur du sujet 
78        */ 
79       @Override 
80       public void setValeur(T v) { 
81           valeur = v; 
82           notifierAbonnés(); 
83       }
 
84    
85       private void notifierAbonnés() { 
86           for (Observateur<T> o : abonnés) { 
87               o.faireMiseAJour(this); 
88           } 
89       } 
90   }
