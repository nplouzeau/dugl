8    public class FabriqueLongueur { 
9     
10       static public Longueur créer(double valeurNumérique, UnitéLongueur u) { 
11           return new LongueurImpl(valeurNumérique, u); 
12       }
 
13    
14       private static class LongueurImpl implements Longueur { 
15           private double valeurEnMètres;
 
16    
17           public LongueurImpl(double valeurNumérique, UnitéLongueur u) { 
18               switch (u) { 
19                   case MÈTRE: 
20                       valeurEnMètres = valeurNumérique; 
21                       break; 
22    
23                   case POUCE: 
24                       valeurEnMètres = valeurNumérique * 2.54E-02; 
25                       break; 
26    
27                   case PIED: 
28                       valeurEnMètres = valeurNumérique * 3.048E-01; 
29                       break; 
30    
31                   case MILE_IMPÉRIAL: 
32                       valeurEnMètres = valeurNumérique * 1.609344E03; 
33    
34                       break; 
35                   case MILE_NAUTIQUE: 
36                       valeurEnMètres = valeurNumérique * 1.852E03; 
37    
38                       break; 
39               } 
40           }
 
41    
42           /** 
43            * Retourne la valeur de la longueur dans une unité de longueur 
44            * 
45            * @param u l'unité du résultat 
46            * @return la valeur numérique de la longueur dans l'unité u 
47            */ 
48           @Override 
49           public double getValeurNumérique(UnitéLongueur u) { 
50               switch (u) { 
51                   case MÈTRE: 
52                       return valeurEnMètres; 
53    
54                   case POUCE: 
55                       return valeurEnMètres / 2.54E-02; 
56    
57                   case PIED: 
58                       return valeurEnMètres / 3.048E-01; 
59    
60                   case MILE_IMPÉRIAL: 
61                       return valeurEnMètres / 1.609344E03; 
62    
63                   case MILE_NAUTIQUE: 
64                       return valeurEnMètres / 1.852E03; 
65               } 
66               return Double.NaN; 
67           }
 
68    
69           /** 
70            * Crée une nouvelle longueur correspondant à la somme de this et de l 
71            * @param l      longueur à ajouter à this 
72            * @return       nouvelle longueur correspond à la somme de this et l 
73            */ 
74           @Override 
75           public Longueur ajouter(Longueur l) { 
76               return new LongueurImpl(this.valeurEnMètres + l.getValeurNumérique(UnitéLongueur.MÈTRE), UnitéLongueur.MÈTRE); 
77           } 
78       } 
79   }
